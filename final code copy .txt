def get_amazon_price(product_name):
    """Get price from Amazon"""
    try:
        headers = {'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'}
        url = f"https://www.amazon.com/s?k={product_name.replace(' ', '+')}"
        response = requests.get(url, headers=headers, timeout=5)
        soup = BeautifulSoup(response.content, 'html.parser')
        
        # Try different selectors for price
        price_elem = soup.find('span', {'class': 'a-price-whole'})
        if price_elem:
            price_text = price_elem.text.replace('$', '').replace(',', '').strip()
            try:
                price = float(price_text.split('.')[0])
                return price, url
            except:
                return None, url
        return None, url
    except Exception as e:
        print(f"Amazon error: {e}")
        return None, f"https://www.amazon.com/s?k={product_name.replace(' ', '+')}"

def get_ebay_price(product_name):
    """Get price from eBay"""
    try:
        headers = {'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'}
        url = f"https://www.ebay.com/sch/i.html?_nkw={product_name.replace(' ', '+')}"
        response = requests.get(url, headers=headers, timeout=5)
        soup = BeautifulSoup(response.content, 'html.parser')
        
        price_elem = soup.find('span', {'class': 's-item__price'})
        if price_elem:
            price_text = price_elem.text.split('$')[-1].strip()
            try:
                price = float(price_text.split()[0].replace(',', ''))
                return price, url
            except:
                return None, url
        return None, url
    except Exception as e:
        print(f"eBay error: {e}")
        return None, f"https://www.ebay.com/sch/i.html?_nkw={product_name.replace(' ', '+')}"

def get_product_prices(product_name):
    """Get prices from multiple websites"""
    prices = {}
    
    # Amazon
    amazon_price, amazon_url = get_amazon_price(product_name)
    prices['Amazon'] = {'price': amazon_price, 'url': amazon_url}
    
    # eBay
    ebay_price, ebay_url = get_ebay_price(product_name)
    prices['eBay'] = {'price': ebay_price, 'url': ebay_url}
    
    # Walmart
    walmart_price, walmart_url = get_walmart_price(product_name)
    prices['Walmart'] = {'price': walmart_price, 'url': walmart_url}
    
    # Google Shopping
    google_url = f"https://www.google.com/shopping/results?q={product_name.replace(' ', '+')}"
    prices['Google Shopping'] = {'price': None, 'url': google_url}
    
    return prices

def load_and_prepare():
    if not os.path.exists(DATA_FILE):
        messagebox.showerror("File error", f"Missing file: {DATA_FILE}")
        return None
    df = pd.read_csv(DATA_FILE)
    drop_cols = ['Product_Container', 'Ship_Mode', 'Customer_Name', 'Customer_Segment', 'Product_Sub-Category']
    df = df.drop(columns=[c for c in drop_cols if c in df.columns])
    if 'Profit' in df.columns:
        df = df[df['Profit'] >= 0].copy()
    if "Order_Priority" in df.columns:
        df["Order_Priority"] = df["Order_Priority"].map(order_priority_mapping).fillna(0).astype(int)
    else:
        df["Order_Priority"] = 0
    if "result" not in df.columns:
        if "Profit" in df.columns:
            med = df["Profit"].median()
            df["result"] = (df["Profit"] > med).astype(int)
        else:
            df["result"] = np.random.randint(0,2,size=len(df))
    for f in ['Sales','Order_Quantity','Profit','Order_Priority']:
        if f not in df.columns:
            df[f] = 0
    return df

class ModelApp:

        root.title("✨ Drop-Shipping AI Predictor")
        root.geometry("900x550")
        root.configure(bg="#0a0e27")
        
        self.setup_styles()
        
        self.df = load_and_prepare()
        if self.df is None:
            root.destroy(); return
        self.X = self.df[['Sales','Order_Quantity','Profit','Order_Priority']]
        self.y = self.df['result']
        self.X_train, self.X_test, self.y_train, self.y_test = train_test_split(self.X, self.y, test_size=0.2, random_state=42)
        self.models = {
            "KNN": KNeighborsClassifier(n_neighbors=5),
            "RandomForest": RandomForestClassifier(n_estimators=100, random_state=42),
            "SVM": SVC(random_state=42)
        }
        self.trained = {}
        self.accuracies = {}
        self.train_all_models()

        self.create_widgets()

    def train_all_models(self):
        for name, model in self.models.items():
            try:
                model.fit(self.X_train, self.y_train)
                preds = model.predict(self.X_test)
                acc = accuracy_score(self.y_test, preds)
                self.trained[name] = model
                self.accuracies[name] = acc
            except Exception:
                self.trained[name] = None
                self.accuracies[name] = None

    def format_acc(self, name):
        a = self.accuracies.get(name)
        return f"✓ {a:.1%}" if a is not None else "N/A"

    def update_acc_text(self):
        self.acc_text.config(state="normal")
        self.acc_text.delete("1.0", tk.END)
        for name in self.models.keys():
            acc = self.format_acc(name)
            self.acc_text.insert(tk.END, f"{name:15} {acc}\n")
        self.acc_text.config(state="disabled")

    def draw_sample_plot(self):
        self.ax.clear()
        names = list(self.accuracies.keys())
        vals = [0 if v is None else v for v in self.accuracies.values()]
        colors = ['#00d4ff', '#00ff88', '#ff006e']
        bars = self.ax.bar(names, vals, color=colors, edgecolor='#1e293b', linewidth=2)
        
        for bar in bars:
            height = bar.get_height()
            self.ax.text(bar.get_x() + bar.get_width()/2., height,
                        f'{height:.1%}', ha='center', va='bottom', color='#e0e7ff', fontweight='bold', fontsize=9)
        
        self.ax.set_ylim(0, 1)
        self.ax.set_ylabel("Accuracy", color='#e0e7ff', fontweight='bold')
        self.ax.tick_params(colors='#a5b4fc')
        self.ax.grid(axis='y', alpha=0.2, color='#1e293b')
        self.fig.tight_layout()
        self.canvas.draw()

    def on_model_change(self):
        self.draw_sample_plot()

    def predict_single(self):
        try:
            s = float(self.entry_sales.get() or 0)
            q = float(self.entry_qty.get() or 0)
            p = float(self.entry_profit.get() or 0)
            pr = order_priority_mapping.get(self.priority_var.get(), 0)
            model_name = self.model_var.get()
            model = self.trained.get(model_name)
            if model is None:
                messagebox.showerror("Model error", f"{model_name} is not trained")
                return
            pred = model.predict(pd.DataFrame({'Sales':[s],'Order_Quantity':[q],'Profit':[p],'Order_Priority':[pr]}))[0]
            txt = "✓ YES - Good Product!" if pred==1 else "✗ NO - Not Suitable"
            acc = self.accuracies.get(model_name)
            acc_txt = f" ({acc:.1%} confidence)" if acc is not None else ""
            color = '#00ff88' if pred==1 else '#ff006e'
            self.lbl_prediction.config(text=f"Prediction: {txt}{acc_txt}", fg=color)
        except Exception as e:
            messagebox.showerror("Input error", str(e))

    def show_conf_matrix(self):
        model_name = self.model_var.get()
        model = self.trained.get(model_name)
        if model is None:
            messagebox.showerror("Model error", f"{model_name} not trained")
            return
        y_pred = model.predict(self.X_test)
        cm = confusion_matrix(self.y_test, y_pred)
        top = tk.Toplevel(self.root)
        top.title(f"🎯 {model_name} - Confusion Matrix")
        top.configure(bg='#0a0e27')
        top.geometry("500x450")
        fig = plt.Figure(figsize=(5, 4), facecolor='#0a0e27', edgecolor='none')
        ax = fig.add_subplot(111)
        ax.set_facecolor('#111827')
        sns.heatmap(cm, annot=True, fmt='d', cmap='cool', cbar=True, ax=ax, 
                   cbar_kws={"label": "Count"}, linewidths=2, linecolor='#1e293b')
        ax.set_xlabel("Predicted", color='#e0e7ff', fontweight='bold')
        ax.set_ylabel("True", color='#e0e7ff', fontweight='bold')
        ax.tick_params(colors='#a5b4fc')
        canvas = FigureCanvasTkAgg(fig, master=top)
        canvas.get_tk_widget().pack(fill="both", expand=True, padx=15, pady=15)
        canvas.draw()

    def export_predictions(self):
        model_name = self.model_var.get()
        model = self.trained.get(model_name)
        if model is None:
            messagebox.showerror("Model error", f"{model_name} not trained")
            return
        raw = pd.read_csv(DATA_FILE)
        if "Order_Priority" in raw.columns:
            raw["Order_Priority"] = raw["Order_Priority"].map(order_priority_mapping).fillna(0).astype(int)
        else:
            raw["Order_Priority"] = 0
        for f in ['Sales','Order_Quantity','Profit','Order_Priority']:
            if f not in raw.columns:
                raw[f]=0
        preds = model.predict(raw[['Sales','Order_Quantity','Profit','Order_Priority']])
        raw["Result"] = ["✓ Yes" if p==1 else "✗ No" for p in preds]
        out = os.path.join(PROJECT_DIR, f"test_with_results_{model_name.lower()}.csv")
        raw.to_csv(out, index=False)
        messagebox.showinfo("✓ Exported", f"Predictions saved!\n\n{out}")

    def show_predicted_products(self):
        model_name = self.model_var.get()
        model = self.trained.get(model_name)
        if model is None:
            messagebox.showerror("Model error", f"{model_name} not trained")
            return

        raw = pd.read_csv(DATA_FILE)
        if "Order_Priority" in raw.columns:
            raw["Order_Priority"] = raw["Order_Priority"].map(order_priority_mapping).fillna(0).astype(int)
        else:
            raw["Order_Priority"] = 0
        for f in ['Sales','Order_Quantity','Profit','Order_Priority']:
            if f not in raw.columns:
                raw[f] = 0

        X_raw = raw[['Sales','Order_Quantity','Profit','Order_Priority']]
        preds = model.predict(X_raw)
        raw['Predicted'] = preds
        
        def product_label(row):
            for c in ['Product_Name', 'Product', 'Product_Sub-Category', 'Product_Sub_Category', 'Product_Category', 'Product Category']:
                if c in row and pd.notna(row[c]) and str(row[c]).strip() != "":
                    return str(row[c])
            if 'Order_ID' in row:
                return f"Order {row['Order_ID']}"
            return "Unknown Product"

        raw['Product_Label'] = raw.apply(product_label, axis=1)
        best = raw[raw['Predicted'] == 1].copy()
        if len(best) > 0:
            for num_col in ['Sales', 'Profit', 'Order_Quantity']:
                if num_col in best.columns:
                    best[num_col] = pd.to_numeric(best[num_col], errors='coerce').fillna(0)
            agg_cols = {}
            if 'Sales' in best.columns: agg_cols['Sales'] = 'sum'
            if 'Profit' in best.columns: agg_cols['Profit'] = 'sum'
            if 'Order_Quantity' in best.columns: agg_cols['Order_Quantity'] = 'sum'
            if 'Order_ID' in best.columns: agg_cols['Order_ID'] = 'first'
            grouped = best.groupby('Product_Label', as_index=False).agg(agg_cols)
            sort_by = ['Profit'] if 'Profit' in grouped.columns else ['Sales']
            grouped = grouped.sort_values(by=sort_by, ascending=False)
        else:
            grouped = pd.DataFrame(columns=['Product_Label','Sales','Profit','Order_Quantity'])

        top = tk.Toplevel(self.root)
        top.title(f"📋 Recommended Products ({model_name})")
        top.geometry("750x500")
        top.configure(bg='#0a0e27')

        header_frame = tk.Frame(top, bg='#1a0033', height=60)
        header_frame.pack(fill="x", padx=0, pady=0)
        header_frame.pack_propagate(False)
        
        title = tk.Label(header_frame, text="✓ ALL RECOMMENDED PRODUCTS", font=("Segoe UI", 12, "bold"), bg='#1a0033', fg='#00d4ff')
        title.pack(pady=8)
        
        info = tk.Label(header_frame, text=f"🤖 Model: {model_name} | 📊 Unique: {len(grouped)} products", bg='#1a0033', fg='#a5b4fc', font=("Segoe UI", 8))
        info.pack(pady=(0, 8))

        cols = ("Product", "Sales", "Profit", "Order_Quantity")
        tree = ttk.Treeview(top, columns=cols, show="headings", height=18)
        for c in cols:
            tree.heading(c, text=c)
            tree.column(c, anchor="center", width=180)
        vsb = ttk.Scrollbar(top, orient="vertical", command=tree.yview)
        tree.configure(yscrollcommand=vsb.set)
        tree.pack(side="left", fill="both", expand=True, padx=(10, 0), pady=10)
        vsb.pack(side="right", fill="y", pady=10, padx=(0, 10))

        tree.tag_configure('best', background='#0f3a1f', foreground='#00ff88')
        tree.tag_configure('row', background='#1f2937', foreground='#a5b4fc')

        for idx, (_, r) in enumerate(grouped.iterrows()):
            prod = r.get('Product_Label', 'Unknown')
            sales = r.get('Sales', '')
            profit = r.get('Profit', '')
            qty = r.get('Order_Quantity', '')
            tag = 'best' if idx % 2 == 0 else 'row'
            tree.insert("", "end", values=(prod, f"{sales}", f"{profit}", f"{qty}"), tags=(tag,))

        if len(grouped) == 0:
            ttk.Label(top, text="❌ No products predicted as best by this model.", foreground="red").pack(padx=10, pady=12)

        def export_best():
            out = os.path.join(PROJECT_DIR, f"predicted_products_{model_name.lower()}.csv")
            grouped.to_csv(out, index=False)
            messagebox.showinfo("✓ Exported", f"Recommended products saved!\n\n{out}")

        ttk.Button(top, text="💾 Export Recommended Products", command=export_best).pack(side="bottom", pady=8)

    def show_top_5_products_links(self):
        model_name = self.model_var.get()
        model = self.trained.get(model_name)
        if model is None:
            messagebox.showerror("Model error", f"{model_name} not trained")
            return

        raw = pd.read_csv(DATA_FILE)
        if "Order_Priority" in raw.columns:
            raw["Order_Priority"] = raw["Order_Priority"].map(order_priority_mapping).fillna(0).astype(int)
        else:
            raw["Order_Priority"] = 0
        for f in ['Sales','Order_Quantity','Profit','Order_Priority']:
            if f not in raw.columns:
                raw[f] = 0

        X_raw = raw[['Sales','Order_Quantity','Profit','Order_Priority']]
        preds = model.predict(X_raw)
        raw['Predicted'] = preds

        def product_label(row):
            for c in ['Product_Name', 'Product', 'Product_Sub-Category', 'Product_Sub_Category', 'Product_Category', 'Product Category']:
                if c in row and pd.notna(row[c]) and str(row[c]).strip() != "":
                    return str(row[c])
            if 'Order_ID' in row:
                return f"Order {row['Order_ID']}"
            return "Unknown Product"

        raw['Product_Label'] = raw.apply(product_label, axis=1)
        best = raw[raw['Predicted'] == 1].copy()
        if len(best) > 0:
            for num_col in ['Sales', 'Profit', 'Order_Quantity']:
                if num_col in best.columns:
                    best[num_col] = pd.to_numeric(best[num_col], errors='coerce').fillna(0)
            agg_cols = {}
            if 'Sales' in best.columns: agg_cols['Sales'] = 'sum'
            if 'Profit' in best.columns: agg_cols['Profit'] = 'sum'
            if 'Order_Quantity' in best.columns: agg_cols['Order_Quantity'] = 'sum'
            if 'Order_ID' in best.columns: agg_cols['Order_ID'] = 'first'
            grouped = best.groupby('Product_Label', as_index=False).agg(agg_cols)
            sort_by = ['Profit'] if 'Profit' in grouped.columns else ['Sales']
            grouped = grouped.sort_values(by=sort_by, ascending=False)

                thread = threading.Thread(target=fetch_prices, args=(prod_name, loading_label, frm_prod), daemon=True)
                thread.start()
